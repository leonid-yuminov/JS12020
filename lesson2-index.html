c = ++a; alert(c); // 2 - потому что, переменная (a) инкрементирована и присвоена значению переменной (c)

d = b++; alert(d); // 1 - потому что, переменная (b) инкрементирована и присвоена значению переменной (d)

c = (2+ ++a); alert(c); // 5 - потому что, переменная (a) инкрементирована (ещё один раз),а также к ней прибавлено число 2, и далее её значение было присвоено переменной (c)

d = (2+ b++); alert(d); // 4 - потому что, переменная (b) инкрементирована (ещё один раз),а также к ней прибавлено число 2, и далее её значение было присвоено переменной (d)

alert(a); // 3 - к переменной (a) было применено две операции инкрементирования
